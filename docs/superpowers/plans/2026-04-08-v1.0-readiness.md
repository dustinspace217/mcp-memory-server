- [ ] **Step 3: Implement schema versioning in sqlite-store.ts**

Replace the migration section in `init()` (lines 140-253) with a version-tracked approach. The key idea:

1. Create tables if they don't exist (same as now)
2. Create `schema_version` table with a single row
3. Detect current version: if `schema_version` doesn't exist, check which columns exist to determine version 1-4
4. Run migrations sequentially for each version bump
5. After all migrations, set version to current

```typescript
// === Schema version tracking ===
// Instead of checking pragma('table_info') for each column, we track a single
// integer version. This makes migrations deterministic and future-proof.
// Version history:
//   1 = base schema (entities, observations, relations)
//   2 = added project column to entities
//   3 = added updated_at, created_at to entities
//   4 = added superseded_at to observations (table rebuild)
//   5 = added created_at, superseded_at to relations (table rebuild) [future: Task 6]

// Create the version table if it doesn't exist
this.db.exec(`
  CREATE TABLE IF NOT EXISTS schema_version (
    version INTEGER NOT NULL
  );
`);

// Read the current version. If the table is empty (first run or existing DB
// without version tracking), detect current state from column existence.
let currentVersion: number;
const versionRow = this.db.prepare('SELECT version FROM schema_version').get() as { version: number } | undefined;

if (versionRow) {
  currentVersion = versionRow.version;
} else {
  // Detect version from existing schema for databases that predate version tracking.
  // Check which columns exist on the entities and observations tables.
  const entityCols = this.db.pragma('table_info(entities)') as { name: string }[];
  const obsCols = this.db.pragma('table_info(observations)') as { name: string }[];

  const hasProject = entityCols.some(c => c.name === 'project');
  const hasUpdatedAt = entityCols.some(c => c.name === 'updated_at');
  const hasSupersededAt = obsCols.some(c => c.name === 'superseded_at');

  if (hasSupersededAt) {
    currentVersion = 4;
  } else if (hasUpdatedAt) {
    currentVersion = 3;
  } else if (hasProject) {
    currentVersion = 2;
  } else if (entityCols.length > 0) {
    currentVersion = 1;
  } else {
    // Fresh database — tables just created with full schema
    currentVersion = 4;
  }

  // Seed the version table so future startups skip detection
  this.db.prepare('INSERT INTO schema_version (version) VALUES (?)').run(currentVersion);
}

// === Run migrations ===

if (currentVersion < 2) {
  // Migration 1 → 2: add project column
  this.db.exec(`
    ALTER TABLE entities ADD COLUMN project TEXT;
    CREATE INDEX IF NOT EXISTS idx_entities_project ON entities(project);
  `);
  this.db.prepare('UPDATE schema_version SET version = 2').run();
  currentVersion = 2;
}

if (currentVersion < 3) {
  // Migration 2 → 3: add entity timestamps
  this.db.transaction(() => {
    this.db.exec(`
      ALTER TABLE entities ADD COLUMN updated_at TEXT NOT NULL DEFAULT '${ENTITY_TIMESTAMP_SENTINEL}';
      ALTER TABLE entities ADD COLUMN created_at TEXT NOT NULL DEFAULT '${ENTITY_TIMESTAMP_SENTINEL}';
    `);
    // Backfill from observation timestamps
    this.db.exec(`
      UPDATE entities SET updated_at = (
        SELECT MAX(created_at) FROM observations
        WHERE entity_id = entities.id AND created_at != 'unknown'
      )
      WHERE updated_at = '${ENTITY_TIMESTAMP_SENTINEL}' AND EXISTS (
        SELECT 1 FROM observations WHERE entity_id = entities.id AND created_at != 'unknown'
      );
    `);
    this.db.exec(`
      UPDATE entities SET created_at = (
        SELECT MIN(created_at) FROM observations
        WHERE entity_id = entities.id AND created_at != 'unknown'
      )
      WHERE created_at = '${ENTITY_TIMESTAMP_SENTINEL}' AND EXISTS (
        SELECT 1 FROM observations WHERE entity_id = entities.id AND created_at != 'unknown'
      );
    `);
  })();
  this.db.prepare('UPDATE schema_version SET version = 3').run();
  currentVersion = 3;
}

if (currentVersion < 4) {
  // Migration 3 → 4: add superseded_at to observations (table rebuild)
  this.db.transaction(() => {
    this.db.exec(`
      CREATE TABLE observations_new (
        id          INTEGER PRIMARY KEY AUTOINCREMENT,
        entity_id   INTEGER NOT NULL REFERENCES entities(id) ON DELETE CASCADE,
        content     TEXT NOT NULL,
        created_at  TEXT NOT NULL,
        superseded_at TEXT NOT NULL DEFAULT '',
        UNIQUE(entity_id, content, superseded_at)
      );
      INSERT INTO observations_new (id, entity_id, content, created_at, superseded_at)
        SELECT id, entity_id, content, created_at, '' FROM observations;
      DROP TABLE observations;
      ALTER TABLE observations_new RENAME TO observations;
    `);
  })();
  this.db.prepare('UPDATE schema_version SET version = 4').run();
  currentVersion = 4;
}
```

```typescript
async entityTimeline(entityName: string, projectId?: string): Promise<EntityTimelineResult | null> {
  const normalizedProject = projectId ?? null;

  // Find the entity by name, optionally filtered by project
  let entitySql = 'SELECT id, name, entity_type, project, updated_at, created_at FROM entities WHERE name = ?';
  const params: (string | null)[] = [entityName];
  if (normalizedProject) {
    entitySql += ' AND (project = ? OR project IS NULL)';
    params.push(normalizedProject);
  }

  const entityRow = this.db.prepare(entitySql).get(...params) as {
    id: number; name: string; entity_type: string;
    project: string | null; updated_at: string; created_at: string;
  } | undefined;

  if (!entityRow) return null;

  // Fetch ALL observations (active AND superseded) sorted chronologically
  const obsRows = this.db.prepare(`
    SELECT content, created_at, superseded_at
    FROM observations WHERE entity_id = ?
    ORDER BY created_at ASC
  `).all(entityRow.id) as { content: string; created_at: string; superseded_at: string }[];

  const observations: TimelineObservation[] = obsRows.map(o => ({
    content: o.content,
    createdAt: o.created_at,
    supersededAt: o.superseded_at,
    status: o.superseded_at === '' ? 'active' as const : 'superseded' as const,
  }));

  // Fetch ALL relations (active AND superseded) involving this entity
  const relRows = this.db.prepare(`
    SELECT from_entity, to_entity, relation_type, created_at, superseded_at
    FROM relations
    WHERE from_entity = ? OR to_entity = ?
    ORDER BY created_at ASC
  `).all(entityName, entityName) as {
    from_entity: string; to_entity: string; relation_type: string;
    created_at: string; superseded_at: string;
  }[];

  const relations: TimelineRelation[] = relRows.map(r => ({
    from: r.from_entity,
    to: r.to_entity,
    relationType: r.relation_type,
    createdAt: r.created_at,
    supersededAt: r.superseded_at,
    status: r.superseded_at === '' ? 'active' as const : 'superseded' as const,
  }));

  return {
    name: entityRow.name,
    entityType: entityRow.entity_type,
    project: entityRow.project,
    createdAt: entityRow.created_at,
    updatedAt: entityRow.updated_at,
    observations,
    relations,
  };
}
```

```typescript
// --- Similarity check: find semantically similar existing observations ---
// After inserting new observations, if the embedding model is ready,
// embed each new observation and find near-neighbors on the same entity.
// Returns matches with cosine similarity > 0.85 so the caller can decide
// whether to supersede, append, or ignore. Best-effort: omitted if model not ready.
if (this.vecTableExists && this.embeddingPipeline.state.status === 'ready') {
  try {
    for (const r of results) {
      if (r.addedObservations.length === 0) continue;

      // Get the entity ID for similarity queries
      const entityRow = findEntity.get(r.entityName) as { id: number } | undefined;
      if (!entityRow) continue;

      const similar: SimilarObservation[] = [];

      for (const obs of r.addedObservations) {
        const embedding = await this.embeddingPipeline.embed(obs.content);
        if (!embedding) continue;

        // KNN search for top-3 nearest neighbors on the same entity
        const knnRows = this.db.prepare(`
          SELECT v.observation_id, v.distance FROM vec_observations v
          WHERE v.embedding MATCH ? AND k = 3
        `).all(
          Buffer.from(embedding.buffer, embedding.byteOffset, embedding.byteLength),
        ) as { observation_id: string; distance: number }[];

        for (const knn of knnRows) {
          // Convert distance to cosine similarity (sqlite-vec uses L2 distance
          // on normalized vectors, so similarity = 1 - distance/2)
          const similarity = 1 - knn.distance / 2;
          if (similarity <= 0.85) continue;

          // Check that this observation belongs to the same entity AND is not
          // the just-inserted observation itself
          const obsRow = this.db.prepare(`
            SELECT o.content FROM observations o
            WHERE o.id = ? AND o.entity_id = ? AND o.superseded_at = '' AND o.content != ?
          `).get(parseInt(knn.observation_id, 10), entityRow.id, obs.content) as { content: string } | undefined;

          if (obsRow) {
            similar.push({ content: obsRow.content, similarity: Math.round(similarity * 1000) / 1000 });
          }
        }
      }

      if (similar.length > 0) {
        r.similarExisting = similar;
      }
    }
  } catch (err: unknown) {
    // Similarity check is best-effort — don't fail the whole operation
    const msg = err instanceof Error ? err.message : String(err);
    console.error(`Similarity check failed: ${msg}`);
  }
}
```

```typescript
process.on('SIGTERM', async () => {
  try {
    if (store && 'shutdown' in store) {
      (store as SqliteStore).shutdown();
    }
    if (store) await store.close();
  } catch (err) {
    console.error('Error closing store during SIGTERM shutdown:', err);
  }
  process.exit(0);
});
```

```typescript
import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import { promises as fs } from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';
import { SqliteStore } from '../sqlite-store.js';

const testDir = path.dirname(fileURLToPath(import.meta.url));

/**
 * Integration tests that load the real embedding model and test end-to-end
 * vector search. These are slow (~10-20s for model load) so skip in CI
 * with SKIP_VECTOR_INTEGRATION=1.
 */
describe('Vector search integration', () => {
  // Skip if env var is set (for fast CI runs)
  if (process.env.SKIP_VECTOR_INTEGRATION === '1') {
    it.skip('skipped via SKIP_VECTOR_INTEGRATION=1', () => {});
    return;
  }

  let store: SqliteStore;
  let storePath: string;

  // Use beforeAll (not beforeEach) so the model loads once for all tests
  beforeAll(async () => {
    storePath = path.join(testDir, `test-vec-integration-${Date.now()}.db`);
    store = new SqliteStore(storePath);
    await store.init();

    // Wait for the model to finish loading (poll with timeout)
    const deadline = Date.now() + 60_000; // 60s timeout for model download
    while (store.vectorState.status === 'loading' && Date.now() < deadline) {
      await new Promise(resolve => setTimeout(resolve, 500));
    }
    if (store.vectorState.status !== 'ready') {
      throw new Error(`Embedding model not ready after 60s: ${store.vectorState.status}`);
    }

    // Wait a bit for the sweep to finish after creating test data
  }, 120_000); // 2 minute timeout for beforeAll

  afterAll(async () => {
    await store.close();
    for (const suffix of ['', '-wal', '-shm']) {
      try { await fs.unlink(storePath + suffix); } catch { /* ignore */ }
    }
  });

  it('should find semantically similar entities via vector search', async () => {
    // Create entities with specific observations
    await store.createEntities([
      { name: 'dog-entity', entityType: 'animal', observations: ['A loyal canine companion that loves to fetch'] },
      { name: 'cat-entity', entityType: 'animal', observations: ['An independent feline that purrs when happy'] },
      { name: 'car-entity', entityType: 'vehicle', observations: ['A four-wheeled motorized vehicle for transport'] },
    ], undefined);

    // Wait for embeddings to be generated
    await new Promise(resolve => setTimeout(resolve, 2000));

    // Search with a semantically related query that wouldn't match LIKE
    const result = await store.searchNodes('puppy pet', undefined);

    // Vector search should find dog-entity even though "puppy" isn't in the observation text
    const entityNames = result.entities.map(e => e.name);
    expect(entityNames).toContain('dog-entity');
  });

  it('should deduplicate LIKE + vector results', async () => {
    await store.createEntities([
      { name: 'dedup-test', entityType: 'test', observations: ['This is a deduplication test observation'] },
    ], undefined);

    await new Promise(resolve => setTimeout(resolve, 2000));

    // Search with a term that matches BOTH LIKE and vector
    const result = await store.searchNodes('deduplication test', undefined);

    // Entity should appear exactly once, not twice
    const matches = result.entities.filter(e => e.name === 'dedup-test');
    expect(matches).toHaveLength(1);
  });

  it('should exclude superseded observations from vector results', async () => {
    await store.createEntities([
      { name: 'supersede-vec-test', entityType: 'test', observations: ['The sky is green'] },
    ], undefined);

    await new Promise(resolve => setTimeout(resolve, 2000));

    // Supersede the observation
    await store.supersedeObservations([
      { entityName: 'supersede-vec-test', oldContent: 'The sky is green', newContent: 'The sky is blue' }
    ]);

    await new Promise(resolve => setTimeout(resolve, 2000));

    // The old observation's embedding should have been removed by syncEmbedding
    // Search should still find the entity (via the new observation)
    const result = await store.searchNodes('sky color', undefined);
    const found = result.entities.find(e => e.name === 'supersede-vec-test');

    if (found) {
      // If found, it should only have the active observation
      const activeObs = found.observations.map(o => o.content);
      expect(activeObs).toContain('The sky is blue');
      expect(activeObs).not.toContain('The sky is green');
    }
    // Not finding it at all is also acceptable (LIKE might not match "sky color")
  });

  it('should return similarExisting for semantically similar observations', async () => {
    await store.createEntities([
      { name: 'sim-check-test', entityType: 'test', observations: ['The user prefers dark mode for all applications'] },
    ], undefined);

    await new Promise(resolve => setTimeout(resolve, 2000));

    // Add a semantically similar observation
    const result = await store.addObservations([
      { entityName: 'sim-check-test', contents: ['Dark theme is the preferred color scheme for apps'] }
    ]);

    // The similarity check should flag the existing observation as similar
    // Note: this is best-effort — may not fire if embeddings aren't ready fast enough
    if (result[0].similarExisting && result[0].similarExisting.length > 0) {
      expect(result[0].similarExisting[0].content).toContain('dark mode');
      expect(result[0].similarExisting[0].similarity).toBeGreaterThan(0.85);
    }
    // Not having similarExisting is acceptable (best-effort)
  });
});
```
